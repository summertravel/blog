CombinedChart

混合图，可以包含我们看到的所有图，这样我们就通过这个来熟悉我们用的LineChart，BarChart。
首先我们的图标是由xy轴以及各种可以自定义添加的线组成，那么我们从xy轴的api开始说起
Y轴：

YAxis leftAxis = mChart.getAxisLeft();//即获得左侧Y轴的实例
    leftAxis.setTypeface(mTfLight);
    leftAxis.setTextColor(ColorTemplate.getHoloBlue());
    leftAxis.setAxisMaximum(200f);
    leftAxis.setAxisMinimum(0f);//设置Y轴的最小值
    leftAxis.setDrawGridLines(true);
    leftAxis.setGranularityEnabled(true);

mChart.getAxisRight();//类似的这是获取右侧Y轴的实例
如果我们不想让他显示
mChart.getAxisRight().setEnabled(false);//设置它的enable为false即可，那么此轴的任何东西都不会绘制了
X轴：

XAxis xAxis = mChart.getXAxis();//这是我们获取x轴的实例
XAxis xAxis = mChart.getXAxis();
xAxis.setTypeface(mTfLight);//设置x轴的字体，一般不用
xAxis.setTextSize(11f);
xAxis.setTextColor(Color.WHITE);
xAxis.setDrawGridLines(false);//设置x轴的轴线是否绘制
xAxis.setDrawAxisLine(false);
xAxis.setValueFormatter(new IAxisValueFormatter() {
    @Override
    public String getFormattedValue(float value, AxisBase axis) {
        return null;
    }
});//设置x轴数据的格式
xAxis.setLabelRotationAngle(-45);//设置x轴字体的旋转        
xAxis.setGranularity(1);//设置x轴的间隔单位，此处设为1，那么我们在上边设置x轴数据的格式的时候value就可以按照下标来取我们设置的值了

Data：这就是我们要设置数据的地方了

像我们的LineChart用的的数据bean就是LineData，

 private void setData(int count, float range) {

    ArrayList<Entry> yVals1 = new ArrayList<Entry>();

    for (int i = 0; i < count; i++) {
        float mult = range / 2f;
        float val = (float) (Math.random() * mult) + 50;
        yVals1.add(new Entry(i, val));
    }

    ArrayList<Entry> yVals2 = new ArrayList<Entry>();

    for (int i = 0; i < count-1; i++) {
        float mult = range;
        float val = (float) (Math.random() * mult) + 450;
        yVals2.add(new Entry(i, val));
//            if(i == 10) {
//                yVals2.add(new Entry(i, val + 50));
//            }
    }

    ArrayList<Entry> yVals3 = new ArrayList<Entry>();

    for (int i = 0; i < count; i++) {
        float mult = range;
        float val = (float) (Math.random() * mult) + 500;
        yVals3.add(new Entry(i, val));
    }

    LineDataSet set1, set2, set3;

    if (mChart.getData() != null &&
            mChart.getData().getDataSetCount() > 0) {
        set1 = (LineDataSet) mChart.getData().getDataSetByIndex(0);
        set2 = (LineDataSet) mChart.getData().getDataSetByIndex(1);
        set3 = (LineDataSet) mChart.getData().getDataSetByIndex(2);
        set1.setValues(yVals1);
        set2.setValues(yVals2);
        set3.setValues(yVals3);
        mChart.getData().notifyDataChanged();
        mChart.notifyDataSetChanged();
    } else {
        // create a dataset and give it a type
        set1 = new LineDataSet(yVals1, "DataSet 1");//此处DataSet 3的标志就是我们在legend中所显示的数据
        set1.setAxisDependency(AxisDependency.LEFT);//设置我们的数据在哪个轴上显示，
        set1.setColor(ColorTemplate.getHoloBlue());//设置折线的颜色
        set1.setCircleColor(Color.WHITE);//设置折线节点的圆的颜色
        set1.setLineWidth(2f);//设置折线的宽度
        set1.setCircleRadius(3f);//设置圆的大小
        set1.setFillAlpha(65);
        set1.setFillColor(ColorTemplate.getHoloBlue());//设置圆内部的颜色
        set1.setHighLightColor(Color.rgb(244, 117, 117));
        set1.setDrawCircleHole(false);
        //set1.setFillFormatter(new MyFillFormatter(0f));
        //set1.setDrawHorizontalHighlightIndicator(false);
        //set1.setVisible(false);
        //set1.setCircleHoleColor(Color.WHITE);

        // create a data object with the datasets
        LineData data = new LineData(set1, set2, set3);
        data.setValueTextColor(Color.WHITE);
        data.setValueTextSize(9f);

        // set data
        mChart.setData(data);
    }
}

BarChart：

    private BarData generateBarData() {
    ArrayList<BarEntry> entries1 = new ArrayList<BarEntry>();
    ArrayList<BarEntry> entries2 = new ArrayList<BarEntry>();

    for (int index = 0; index < itemcount; index++) {
        entries1.add(new BarEntry(0, getRandom(25, 25)));//0是第0组的数据
        // stacked
        entries2.add(new BarEntry(0, new float[]{getRandom(13, 12), getRandom(13, 12)}));
    }

    BarDataSet set1 = new BarDataSet(entries1, "Bar 1");
    set1.setColor(Color.rgb(60, 220, 78));
    set1.setValueTextColor(Color.rgb(60, 220, 78));
    set1.setValueTextSize(10f);
    set1.setAxisDependency(YAxis.AxisDependency.LEFT);
    set1.setDrawValues(false);//设置是否柱状（折线）上是否显示数据

    BarDataSet set2 = new BarDataSet(entries2, "");
    set2.setStackLabels(new String[]{"Stack 1", "Stack 2"});
    set2.setColors(new int[]{Color.rgb(61, 165, 255), Color.rgb(23, 197, 255)});
    set2.setValueTextColor(Color.rgb(61, 165, 255));
    set2.setValueTextSize(10f);
    set2.setAxisDependency(YAxis.AxisDependency.LEFT);

    float groupSpace = 0.06f; //组和组之间的距离
    float barSpace = 0.02f; // x2 dataset 每组之间bar的距离
    float barWidth = 0.45f; // x2 dataset bar的宽度
    // (0.45 + 0.02) * 2 + 0.06 = 1.00 -> interval per "group"

    BarData d = new BarData(set1, set2);
    d.setBarWidth(barWidth);

    // make this BarData object grouped
    d.groupBars(0, groupSpace, barSpace); // start at x = 0

    return d;
}

总结：如果我们设置xy轴的样式就拿到xy轴实例去设置，如果对数据的展示样式有要求，就在Dataset去设置我们需要的api。
